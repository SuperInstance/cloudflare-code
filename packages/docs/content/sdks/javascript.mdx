## Quick Start

```typescript
import { ClaudeFlare } from '@claudeflare/sdk'

const client = new ClaudeFlare({
  apiKey: process.env.CLAUDEFLARE_API_KEY,
  baseURL: 'https://api.claudeflare.com'
})

const response = await client.chat.completions.create({
  messages: [
    { role: 'user', content: 'Hello, ClaudeFlare!' }
  ]
})

console.log(response.choices[0].message.content)
```

## Configuration

### Client Options

```typescript
const client = new ClaudeFlare({
  apiKey: 'your-api-key',           // Required
  baseURL: 'https://api.claudeflare.com', // Optional
  timeout: 30000,                    // Optional: 30 seconds
  maxRetries: 3,                     // Optional: retry attempts
  headers: {                         // Optional: custom headers
    'X-Custom-Header': 'value'
  }
})
```

### Streaming Completion

```typescript
const stream = await client.chat.completions.stream({
  messages: [
    { role: 'user', content: 'Tell me a story' }
  ]
})

for await (const chunk of stream) {
  const content = chunk.choices[0]?.delta?.content
  if (content) {
    process.stdout.write(content)
  }
```

## Error Handling

### Try-Catch

```typescript
try {
  const response = await client.chat.completions.create({
    messages: [{ role: 'user', content: 'Hello!' }]
  })
} catch (error) {
  if (error instanceof ClaudeFlareError) {
    console.error('API Error:', error.message)
    console.error('Status:', error.status)
    console.error('Code:', error.code)
  } else {
    console.error('Unknown error:', error)
  }
}
```

### Custom Types

```typescript
interface CodeCompletionRequest {
  code: string
  language: string
  maxTokens?: number
}

async function completeCode(request: CodeCompletionRequest): Promise<string> {
  const response = await client.chat.completions.create({
    messages: [
      {
        role: 'system',
        content: `You are a ${request.language} expert. Complete the code.`
      },
      {
        role: 'user',
        content: request.code
      }
    ],
    max_tokens: request.maxTokens || 500,
    temperature: 0.2
  })

  return response.choices[0].message.content
}
```

## Advanced Usage

### Webhook Handler

```typescript
import { ClaudeFlare, verifyWebhookSignature } from '@claudeflare/sdk'

export async function POST(request: Request) {
  const signature = request.headers.get('X-ClaudeFlare-Signature')
  const body = await request.text()

  if (!verifyWebhookSignature(body, signature, process.env.WEBHOOK_SECRET!)) {
    return new Response('Invalid signature', { status: 401 })
  }

  const event = JSON.parse(body)

  switch (event.type) {
    case 'completion.completed':
      console.log('Completion:', event.data)
      break
    case 'agent.task_completed':
      console.log('Agent task:', event.data)
      break
  }

  return new Response('OK')
}
```

### Custom Middleware

```typescript
import { ClaudeFlare } from '@claudeflare/sdk'

const client = new ClaudeFlare({
  apiKey: process.env.CLAUDEFLARE_API_KEY!
})

// Add request interceptor
client.use({
  async request(config) {
    console.log('Request:', config.method, config.url)
    return config
  }
})

// Add response interceptor
client.use({
  async response(response) {
    console.log('Response:', response.status)
    return response
  }
})
```

## Examples

### Express.js Integration

```typescript
import express from 'express'
import { ClaudeFlare } from '@claudeflare/sdk'

const app = express()
const client = new ClaudeFlare({
  apiKey: process.env.CLAUDEFLARE_API_KEY!
})

app.post('/api/chat', async (req, res) => {
  try {
    const { messages } = req.body

    const response = await client.chat.completions.create({
      messages
    })

    res.json(response)
  } catch (error) {
    res.status(500).json({ error: error.message })
  }
})

app.listen(3000)
```

### Next.js Integration

```typescript
// app/api/chat/route.ts
import { ClaudeFlare } from '@claudeflare/sdk'
import { NextResponse } from 'next/server'

const client = new ClaudeFlare({
  apiKey: process.env.CLAUDEFLARE_API_KEY!
})

export async function POST(request: Request) {
  const { messages } = await request.json()

  const response = await client.chat.completions.create({
    messages
  })

  return NextResponse.json(response)
}
```

# Build Your First Project

Let's build an AI-powered code completion tool using ClaudeFlare. This project will demonstrate the core features of the platform.

## What We'll Build

We'll create a code completion service that:
- Accepts partial code snippets
- Returns intelligent completions
- Caches results for performance
- Handles streaming responses

  <Tab label="JavaScript">
    ```javascript
    import { ClaudeFlare } from '@claudeflare/sdk'

    // Initialize the client
    const client = new ClaudeFlare({
      apiKey: process.env.CLAUDEFLARE_API_KEY,
      baseURL: process.env.CLAUDEFLARE_URL || 'http://localhost:8787'
    })

    /**
     * Get code completion
     * @param {string} code - Partial code snippet
     * @param {string} language - Programming language
     * @returns {Promise<string>} - Completed code
     */
    async function getCodeCompletion(code, language = 'javascript') {
      try {
        const systemPrompt = `You are an expert ${language} programmer.
        Complete the given code snippet with only the necessary code.
        Return only the completion, no explanations.`

        const response = await client.chat.completions.create({
          messages: [
            { role: 'system', content: systemPrompt },
            { role: 'user', content: `Complete this code:\n\`\`\`${language}\n${code}\n\`\`\`` }
          ],
          model: 'auto',
          temperature: 0.2, // Lower temperature for more deterministic completions
          max_tokens: 500
        })

        return response.choices[0].message.content
      } catch (error) {
        console.error('Error getting completion:', error)
        throw error
      }
    }

    /**
     * Get streaming code completion
     * @param {string} code - Partial code snippet
     * @param {string} language - Programming language
     * @returns {AsyncGenerator<string>} - Streaming completion
     */
    async function* getStreamingCompletion(code, language = 'javascript') {
      try {
        const systemPrompt = `You are an expert ${language} programmer.
        Complete the given code snippet with only the necessary code.
        Return only the completion, no explanations.`

        const stream = await client.chat.completions.stream({
          messages: [
            { role: 'system', content: systemPrompt },
            { role: 'user', content: `Complete this code:\n\`\`\`${language}\n${code}\n\`\`\`` }
          ],
          model: 'auto',
          temperature: 0.2,
          stream: true
        })

        for await (const chunk of stream) {
          const content = chunk.choices[0]?.delta?.content
          if (content) {
            yield content
          }
        }
      } catch (error) {
        console.error('Error in streaming completion:', error)
        throw error
      }
    }

    // Export functions
    export { getCodeCompletion, getStreamingCompletion }

    // Example usage
    if (import.meta.url === `file://${process.argv[1]}`) {
      const code = 'function add(a, b) {\n  return'

      console.log('Non-streaming completion:')
      const completion = await getCodeCompletion(code)
      console.log(completion)

      console.log('\nStreaming completion:')
      for await (const chunk of getStreamingCompletion(code)) {
        process.stdout.write(chunk)
      }
      console.log('\n')
    }
    ```

### Step 3: Create a Web Server

Create `src/server.js`:

```javascript
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import { getCodeCompletion, getStreamingCompletion } from './index.js'

const app = new Hono()

// Enable CORS
app.use('*', cors())

// Health check
app.get('/', (c) => {
  return c.json({ status: 'ok', service: 'code-completion' })
})

// Code completion endpoint
app.post('/complete', async (c) => {
  try {
    const { code, language, options } = await c.req.json()

    if (!code) {
      return c.json({ error: 'Code is required' }, 400)
    }

    const completion = await getCodeCompletion(
      code,
      language || 'javascript',
      options
    )

    return c.json({
      code,
      completion,
      full: code + completion
    })
  } catch (error) {
    return c.json(
      { error: error.message },
      500
    )
  }
})

// Streaming completion endpoint
app.post('/complete/stream', async (c) => {
  try {
    const { code, language, options } = await c.req.json()

    if (!code) {
      return c.json({ error: 'Code is required' }, 400)
    }

    const encoder = new TextEncoder()
    const stream = new ReadableStream({
      async start(controller) {
        try {
          for await (const chunk of getStreamingCompletion(
            code,
            language || 'javascript',
            options
          )) {
            controller.enqueue(encoder.encode(`data: ${JSON.stringify({ chunk })}\n\n`))
          }
          controller.enqueue(encoder.encode('data: [DONE]\n\n'))
          controller.close()
        } catch (error) {
          controller.error(error)
        }
      }
    })

    return new Response(stream, {
      headers: {
        'Content-Type': 'text/event-stream',
        'Cache-Control': 'no-cache',
        'Connection': 'keep-alive'
      }
    })
  } catch (error) {
    return c.json(
      { error: error.message },
      500
    )
  }
})

export default app
```

```html
    const API_URL = 'http://localhost:3000';

    async function completeCode() {
      const code = document.getElementById('codeInput').value;
      const language = document.getElementById('languageSelect').value;
      const output = document.getElementById('output');
      const completion = document.getElementById('completion');
      const btn = document.getElementById('completeBtn');

      if (!code.trim()) {
        alert('Please enter some code');
        return;
      }

      btn.disabled = true;
      output.style.display = 'block';
      completion.textContent = 'Loading...';

      try {
        const response = await fetch(`${API_URL}/complete`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ code, language })
        });

        const data = await response.json();
        completion.textContent = data.completion || 'No completion available';
      } catch (error) {
        completion.textContent = 'Error: ' + error.message;
      } finally {
        btn.disabled = false;
      }
    }

    async function streamCompletion() {
      const code = document.getElementById('codeInput').value;
      const language = document.getElementById('languageSelect').value;
      const output = document.getElementById('output');
      const completion = document.getElementById('completion');
      const btn = document.getElementById('streamBtn');

      if (!code.trim()) {
        alert('Please enter some code');
        return;
      }

      btn.disabled = true;
      output.style.display = 'block';
      completion.textContent = '';

      try {
        const response = await fetch(`${API_URL}/complete/stream`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ code, language })
        });

        const reader = response.body.getReader();
        const decoder = new TextDecoder();

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          const text = decoder.decode(value);
          const lines = text.split('\n');

          for (const line of lines) {
            if (line.startsWith('data: ')) {
              const data = line.slice(6);
              if (data === '[DONE]') continue;

              try {
                const parsed = JSON.parse(data);
                completion.textContent += parsed.chunk || '';
              } catch (e) {
                // Skip invalid JSON
              }
            }
          }
        }
      } catch (error) {
        completion.textContent = 'Error: ' + error.message;
      } finally {
        btn.disabled = false;
      }
    }
  </script>
```

### Step 5: Add Testing

Create `tests/completion.test.js`:

```javascript
import { describe, it, expect } from 'vitest'
import { getCodeCompletion } from '../src/index.js'

describe('Code Completion', () => {
  it('should complete a simple function', async () => {
    const code = 'function greet(name) {\n  return'
    const completion = await getCodeCompletion(code, 'javascript')

    expect(completion).toBeTruthy()
    expect(completion.length).toBeGreaterThan(0)
    expect(completion.toLowerCase()).toContain('hello')
  })

  it('should handle different languages', async () => {
    const tests = [
      { code: 'def add(a, b):', language: 'python' },
      { code: 'func add(a, b int)', language: 'go' },
      { code: 'interface User', language: 'typescript' }
    ]

    for (const test of tests) {
      const completion = await getCodeCompletion(test.code, test.language)
      expect(completion).toBeTruthy()
    }
  })

  it('should respect temperature option', async () => {
    const code = 'const data ='

    const lowTemp = await getCodeCompletion(code, 'javascript', { temperature: 0.1 })
    const highTemp = await getCodeCompletion(code, 'javascript', { temperature: 1.0 })

    expect(lowTemp).toBeTruthy()
    expect(highTemp).toBeTruthy()
  })
})
```

   ```javascript
   async function getCachedCompletion(code, language) {
     const cacheKey = `${language}:${hash(code)}`

     // Check cache
     const cached = await cache.get(cacheKey)
     if (cached) return cached

     // Get completion
     const completion = await getCodeCompletion(code, language)

     // Store in cache
     await cache.set(cacheKey, completion, { ttl: 3600 })

     return completion
   ```

2. **Add Multi-Agent Workflow**
   ```javascript
   async function completeWithReview(code, language) {
     // Get completion
     const completion = await getCodeCompletion(code, language)

     // Review completion
     const review = await client.agents.orchestrate({
       task: 'Review this code',
       agents: ['reviewer'],
       input: { code: code + completion }
     })

     return { completion, review }
   }
   ```

3. **Add Metrics**
   ```javascript
   async function trackCompletion(code, language) {
     const start = Date.now()

     try {
       const completion = await getCodeCompletion(code, language)
       const latency = Date.now() - start

       await metrics.track('completion_success', {
         language,
         latency,
         length: completion.length
       })

       return completion
     } catch (error) {
       await metrics.track('completion_error', {
         language,
         error: error.message
       })
       throw error
     }
   }
   ```

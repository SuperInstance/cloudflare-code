# Error Handling

Build robust applications with proper error handling, retry logic, and graceful degradation when using ClaudeFlare APIs.

## Overview

ClaudeFlare APIs can return errors for various reasons:

- Rate limiting
- Provider unavailability
- Invalid requests
- Network issues
- Timeouts

Proper error handling ensures your application remains functional and provides good user experience.

## Error Types

### 1. Rate Limit Errors

Status: 429

```typescript
{
```

## Best Practices

### 1. Implement Exponential Backoff

Retry failed requests with increasing delays:

```typescript
async function retryWithBackoff<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      if (attempt === maxRetries - 1) throw error;

      // Exponential backoff: 1s, 2s, 4s, 8s...
      const delay = Math.pow(2, attempt) * 1000;
      console.log(`Retry ${attempt + 1}/${maxRetries} after ${delay}ms`);
      await sleep(delay);
    }
  }
  throw new Error('Max retries exceeded');
}

// Usage
const result = await retryWithBackoff(() =>
  client.chat.completions.create(request)
);
```

### 2. Handle Specific Error Codes

```typescript
try {
  const result = await client.chat.completions.create(request);
} catch (error) {
  switch (error.code) {
    case 'RATE_LIMITED':
      // Implement backoff or queue
      await handleRateLimit(error.retryAfter);
      break;

    case 'QUOTA_EXCEEDED':
      // Notify user or upgrade plan
      await notifyQuotaExceeded();
      break;

    case 'PROVIDER_UNAVAILABLE':
      // Try different provider
      const result = await client.chat.completions.create({
        ...request,
        provider: 'groq'
      });
      break;

    case 'WORKFLOW_TIMEOUT':
      // Retry with longer timeout
      const result = await client.agents.orchestrate({
        ...request,
        timeout: 600 // 10 minutes
      });
      break;

    default:
      // Log and handle unexpected errors
      console.error('Unexpected error:', error);
      throw error;
  }
}
```

### 4. Implement Graceful Degradation

```typescript
async function generateCodeWithFallback(prompt: string) {

    // Try primary approach
```

### 5. Queue Requests for Rate Limits

```typescript
class RequestQueue {
  private queue: Array<() => Promise<any>> = [];
  private processing = false;
  private delay = 1000; // 1 second between requests

  async add<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      this.queue.push(async () => {
        try {
          resolve(await fn());
        } catch (error) {
          reject(error);
        }
      });
      this.process();
    });
  }

  private async process() {
    if (this.processing || this.queue.length === 0) return;

    this.processing = true;
    while (this.queue.length > 0) {
      const fn = this.queue.shift();
      await fn();
      await sleep(this.delay);
    }
    this.processing = false;
  }
}

// Usage
const queue = new RequestQueue();
const result = await queue.add(() =>
  client.chat.completions.create(request)
);
```

```typescript
    }

}
```

## Retry Strategies

### Fixed Delay Retry

```typescript
async function fixedDelayRetry<T>(
  fn: () => Promise<T>,
  retries = 3,
  delay = 1000
): Promise<T> {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (i === retries - 1) throw error;
      await sleep(delay);
    }
  }
  throw new Error('Max retries exceeded');
}
```

### Exponential Backoff with Jitter

```typescript
async function exponentialBackoffRetry<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      if (attempt === maxRetries - 1) throw error;

      // Exponential backoff with jitter
      const baseDelay = Math.pow(2, attempt) * 1000;
      const jitter = Math.random() * 1000;
      const delay = baseDelay + jitter;

      await sleep(delay);
    }
  }
  throw new Error('Max retries exceeded');
}
```

### Circuit Breaker Pattern

```typescript
class CircuitBreaker {
  private failures = 0;
  private lastFailureTime = 0;
  private state: 'closed' | 'open' | 'half-open' = 'closed';
  private threshold = 5;
  private timeout = 60000; // 1 minute

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === 'open') {
      if (Date.now() - this.lastFailureTime > this.timeout) {
        this.state = 'half-open';
      } else {
        throw new Error('Circuit breaker is open');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess() {
    this.failures = 0;
    this.state = 'closed';
  }

  private onFailure() {
    this.failures++;
    this.lastFailureTime = Date.now();

    if (this.failures >= this.threshold) {
      this.state = 'open';
    }
  }
}

// Usage
const breaker = new CircuitBreaker();
const result = await breaker.execute(() =>
  client.chat.completions.create(request)
);
```

### Track Error Rates

```typescript
class ErrorTracker {
  private errors = new Map<string, number>();
  private window = 60000; // 1 minute
  private timestamps: Map<string, number[]> = new Map();

  record(code: string) {
    const now = Date.now();
    const timestamps = this.timestamps.get(code) || [];

    // Remove old timestamps
    const recent = timestamps.filter(t => now - t < this.window);
    recent.push(now);

    this.timestamps.set(code, recent);
    this.errors.set(code, recent.length);
  }

  getRate(code: string): number {
    const count = this.errors.get(code) || 0;
    return count / (this.window / 1000); // errors per second
  }

  shouldTriggerAlert(code: string, threshold = 10): boolean {
    return this.getRate(code) > threshold;
  }
}
```

### Degraded Mode

```typescript
class DegradedModeHandler {
  private errorCount = 0;
  private threshold = 10;
  private degraded = false;

  async handle(request: ChatRequest) {
    try {
      const result = await client.chat.completions.create(request);
      this.onSuccess();
      return result;
    } catch (error) {
      this.onError();

      if (this.degraded) {
        return this.handleDegraded(request);
      }

      throw error;
    }
  }

  private onSuccess() {
    this.errorCount = 0;
    this.degraded = false;
  }

  private onError() {
    this.errorCount++;
    if (this.errorCount >= this.threshold) {
      this.degraded = true;
      console.warn('Entering degraded mode');
    }
  }

  private async handleDegraded(request: ChatRequest) {
    // Return cached response if available
    const cached = await cache.get(request);
    if (cached) return cached;

    // Return simplified response
    return {
      message: 'Service temporarily unavailable',
      suggestions: ['Try again later', 'Contact support']
    };
  }
}
```
